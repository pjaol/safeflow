```swift
import Foundation

extension Array where Element == Int {
    // Helper to calculate weighted average
    func weightedAverage() -> Double {
        let weights = self.indices.map { Double($0 + 1) }
        let weightedSum = zip(self, weights).reduce(0.0) { $0 + Double($1.0) * $1.1 }
        let totalWeight = weights.reduce(0.0, +)
        return weightedSum / totalWeight
    }
}

class CyclePredictionHelper {
    static func calculateCycleLengths(periodStarts: [Date]) -> [Int] {
        // Calculate lengths between consecutive cycles
        return zip(periodStarts, periodStarts.dropFirst())
            .map { Calendar.current.dateComponents([.day], from: $0, to: $1).day ?? 0 }
    }
    
    static func imputeMissingPeriods(periodStarts: [Date], maxGap: Int = 45) -> [Date] {
        // Impute missing period starts if the gap exceeds the max gap
        guard periodStarts.count >= 2 else { return periodStarts }
        
        var imputedPeriods = [periodStarts.first!]
        for i in 1..<periodStarts.count {
            let previous = imputedPeriods.last!
            let next = periodStarts[i]
            let gap = Calendar.current.dateComponents([.day], from: previous, to: next).day ?? 0
            
            if gap > maxGap {
                let imputedDate = Calendar.current.date(byAdding: .day, value: maxGap, to: previous)!
                imputedPeriods.append(imputedDate)
            }
            imputedPeriods.append(next)
        }
        return imputedPeriods
    }
    
    static func movingAverage(cycles: [Int], windowSize: Int) -> Double? {
        // Use a moving average for smoothing cycles
        guard cycles.count >= windowSize else { return nil }
        let recentCycles = Array(cycles.suffix(windowSize))
        return Double(recentCycles.reduce(0, +)) / Double(recentCycles.count)
    }
}
```

Integrate the helper functions into the `CycleStore` class.

```swift
@MainActor
class CycleStore: ObservableObject {
    @Published private(set) var cycleDays: [CycleDay] = []
    
    func predictNextPeriod() -> Date? {
        var periodStarts = getPeriodStarts()
        
        // Impute missing periods
        periodStarts = CyclePredictionHelper.imputeMissingPeriods(periodStarts: periodStarts)
        guard periodStarts.count >= 2 else { return nil }
        
        // Calculate cycle lengths
        let cycles = CyclePredictionHelper.calculateCycleLengths(periodStarts: periodStarts)
        
        // Use weighted moving average
        let windowSize = 3
        guard let averageCycleLength = CyclePredictionHelper.movingAverage(cycles: cycles, windowSize: windowSize) else {
            return nil
        }
        
        // Predict the next period
        guard let lastPeriodStart = periodStarts.last else { return nil }
        return Calendar.current.date(byAdding: .day, value: Int(round(averageCycleLength)), to: lastPeriodStart)
    }
    
    private func getPeriodStarts() -> [Date] {
        return cycleDays
            .filter { $0.flow != nil } // Use flow data to detect period days
            .map { $0.date }
            .sorted()
    }
}
```
